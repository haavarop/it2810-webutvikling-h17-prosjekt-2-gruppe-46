Email verification


function validate() {
    $("#result").text("").fadeIn();
    var email = $("#email-field").val();

    if(validateEmailSubmit(email)){
        $("#result")
            .text(email + " added to newsletter")
            .css("color", "green");
        window.setTimeout(wait,3000);
    }else{
        $("#result")
            .text(email + " is not valid")
            .css("color", "red");
        window.setTimeout(wait,3000);

    }
    return false;
}

function wait() {
   $("#result").fadeOut("slow");
}
